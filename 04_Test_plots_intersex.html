2. Plots of common periwinkle (strandsnegl)

One station, all years

station <- "6"
data_strandsnegl %>%
  filter(!is.na(ISI)) %>%
  filter(Station %in% station) %>%
  ggplot(aes(ISI)) + 
  geom_histogram(binwidth = 1) +
  facet_wrap(~Year) +
  labs(title = paste("Common periwinkle, station", station))

One year, all stations

year <- 2007

data_3sp <- bind_rows(
